create getTopRatedMovies class to take a reference from a domain layer BaseMoviesRepositories, then call the abstract method related to this class from it .

class GetTopRatedMovies{

    final BaseMoviesRepositories baseMoviesRepositories;
   GetTopRatedMovies(this.baseMoviesRepositories);

   Future<List<Movie>> execute() async {
        return await baseMoviesRepositories.getTopRatedMovies();
    }
}
