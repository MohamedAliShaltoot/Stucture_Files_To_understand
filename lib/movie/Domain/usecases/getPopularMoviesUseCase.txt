create getPopularMovies class to take a reference from a domain layer BaseMoviesRepositories, then call the abstract method related to this class from it .

class GetPopularMovies{

    final BaseMoviesRepositories baseMoviesRepositories;
   GetPopularMovies(this.baseMoviesRepositories);

   Future<List<Movie>> execute() async {
        return await baseMoviesRepositories.getPopularMovies();
    }
}
